<script>
// Panel switching — one chip bar, no duplicates
const panels = {
  all:'all-active', nba:'nba-active', wnba:'wnba-active',
  ncaam:'ncaam-active', ncaaw:'ncaaw-active'
};

function showPanel(id, el) {
  // Hide all
  document.querySelectorAll('.panel').forEach(p=>p.classList.remove('active'));
  document.getElementById('panel-'+id).classList.add('active');
  // Reset chips
  document.querySelectorAll('.lc').forEach(c=>{
    c.className='lc';
  });
  // Activate clicked chip
  el.classList.add(panels[id]);
}

// Bracket tab switching
document.querySelectorAll('.bt').forEach(bt=>{
  bt.addEventListener('click',()=>{
    bt.closest('.bracket-tabs').querySelectorAll('.bt').forEach(b=>b.classList.remove('active'));
    bt.classList.add('active');
  });
});

// WNBA countdown to May 16 2026
function updateCountdown(){
  const target = new Date('2026-05-16T19:00:00');
  const now = new Date();
  const diff = target - now;
  if(diff<=0){document.getElementById('days-count').textContent='LIVE';return;}
  const d=Math.floor(diff/86400000);
  const h=Math.floor((diff%86400000)/3600000);
  const m=Math.floor((diff%3600000)/60000);
  const s=Math.floor((diff%60000)/1000);
  document.getElementById('days-count').textContent=String(d).padStart(2,'0');
  document.getElementById('hrs-count').textContent=String(h).padStart(2,'0');
  document.getElementById('min-count').textContent=String(m).padStart(2,'0');
  document.getElementById('sec-count').textContent=String(s).padStart(2,'0');
}
updateCountdown();
setInterval(updateCountdown,1000);
</script>
