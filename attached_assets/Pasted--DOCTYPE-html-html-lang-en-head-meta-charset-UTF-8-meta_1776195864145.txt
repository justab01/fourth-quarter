function updateClock(s) {
  const m = Math.floor(s/60);
  const sec = (s%60).toString().padStart(2,'0');
  document.getElementById('gameClock').textContent = `${m}:${sec}`;

function updateWP(okc) {
  document.getElementById('wpOKC').textContent = okc + '%';
  document.getElementById('wpHOU').textContent = (100-okc) + '%';
  document.getElementById('wpBarOKC').style.width = okc + '%';
  document.getElementById('wpBarHOU').style.width = (100-okc) + '%';
}

function addPlay(p) {
  const diff = okcScore - houScore;
  const score = `${okcScore}–${houScore}`;
  const el = document.createElement('div');
  el.className = `play-item ${p.pts > 0 ? 'score-play' : ''}`;

      <div class="play-desc">${p.desc}</div>

function simulatePlay() {
  if (!simRunning || playIdx >= newPlays.length) {
    simRunning = false;
    return;
  }
  const p = newPlays[playIdx++];
  seconds = Math.max(0, seconds - Math.floor(Math.random()*20+8));
  updateClock(seconds);

  if (p.team === 'okc') {
    okcScore += p.pts;
    wpOKC = Math.min(65, wpOKC + (p.pts > 0 ? (p.pts === 3 ? 12 : 7) : 2));
  } else {
    houScore += p.pts;
    wpOKC = Math.max(20, wpOKC - (p.pts > 0 ? 6 : 0));
  }

  document.getElementById('scoreOKC').textContent = okcScore;
  document.getElementById('scoreHOU').textContent = houScore;
  updateWP(wpOKC);
  addPlay(p);

  const diff = okcScore - houScore;
  let banner = '🏀 ';
  if (diff === 0) banner += 'TIE GAME — ANYTHING CAN HAPPEN';
  else if (diff > 0) banner += `OKC LEADS BY ${diff} · Q4`;
  else banner += `HOU LEADS BY ${Math.abs(diff)} · Q4`;
  document.querySelector('.context-banner').innerHTML = '🔥 ' + banner.substring(2);

  if (playIdx % 2 === 0) updateAI();

  // Update momentum slightly
  momentumData.push(diff);
  if (momentumData.length > 100) momentumData.shift();
  drawMomentum();

  setTimeout(simulatePlay, 3500 + Math.random()*2000);
}
